## 🎯 Icon

### Назначение

**Domain модель** — представляет иконку в бизнес-логике приложения.

### Исходный Код

```swift
import Foundation

/// Представляет иконку в приложении.
/// 
/// Domain модель, используется в бизнес-логике.
/// Содержит все данные об иконке: от имени до метаданных.
struct Icon {
    /// Полное имя иконки в формате "collection:icon_name"
    /// Пример: "mdi:home", "fa:user"
    let fullName: String
    
    /// Теги для поиска
    /// Пример: ["home", "house", "building"]
    var tags: [String] = []
    
    /// Ширина иконки в пикселях
    var width: Int = 0
    
    /// Высота иконки в пикселях
    var height: Int = 0
    
    /// Префикс коллекции (часть до ':')
    /// Пример: "mdi" из "mdi:home"
    var collectionPrefix: String {
        (fullName.split(separator: ":").first).map(String.init) ?? ""
    }
    
    /// Имя иконки (часть после ':')
    /// Пример: "home" из "mdi:home"
    var iconName: String {
        (fullName.split(separator: ":").last).map(String.init) ?? ""
    }
}
```

### Ключевые Моменты

| Момент | Описание |
|--------|----------|
| **Value Type** | Struct — копируется при присваивании |
| **Immutable fullName** | Нельзя изменить после создания |
| **Mutable metadata** | tags, width, height можно обновить |
| **Computed Properties** | collectionPrefix и iconName вычисляются |

### Использование

```swift
// Создание
let icon = Icon(fullName: "mdi:home")

// Обновление метаданных
var icon = Icon(fullName: "mdi:home")
icon.tags = ["house", "building"]
icon.width = 24
icon.height = 24

// Вычисляемые свойства
print(icon.collectionPrefix)  // "mdi"
print(icon.iconName)          // "home"
```

## 📄 IconDetailResponse

### Назначение

**Data модель** — точное соответствие JSON от API для деталей иконок.

### Исходный Код

```swift
import Foundation

/// Ответ API с деталями иконок коллекции.
/// 
/// Data модель для декодирования JSON.
/// Соответствует структуре: https://api.iconify.design/{prefix}.json
struct IconDetailResponse: Codable {
    /// Префикс коллекции
    /// Пример: "mdi"
    let prefix: String
    
    /// Ширина иконок по умолчанию
    let width: Int
    
    /// Высота иконок по умолчанию
    let height: Int
    
    /// Словарь иконок: имя → метаданные
    let icons: [String: IconMetadata]
}

/// Метаданные отдельной иконки.
struct IconMetadata: Codable {
    /// Теги иконки (опционально)
    let tags: [String]?
    
    /// SVG path data
    let body: String
}
```

### Пример JSON

```json
{
  "prefix": "mdi",
  "width": 24,
  "height": 24,
  "icons": {
    "home": {
      "tags": ["house", "building"],
      "body": "M10 20v-6h4v6h5v-8h3L12 3 2 12h3v8z"
    },
    "user": {
      "tags": ["person", "account"],
      "body": "M12 12c2.21 0 4-1.79 4-4s-1.79-4-4-4-4 1.79-4 4 1.79 4 4 4zm0 2c-2.67 0-8 1.34-8 4v2h16v-2c0-2.66-5.33-4-8-4z"
    }
  }
}
```

## 🔍 SearchResult

### Назначение

**Data модель** — результат поиска иконок.

### Исходный Код

```swift
import Foundation

/// Результат поиска иконок.
/// 
/// Data модель для декодирования JSON поиска.
/// Соответствует структуре: https://api.iconify.design/search?query=...
struct SearchResult: Codable {
    /// Список найденных иконок (полные имена)
    /// Пример: ["mdi:home", "fa:home"]
    let icons: [String]
    
    /// Общее количество найденных иконок
    let total: Int
    
    /// Информация о коллекциях
    let collections: [String: CollectionInfo]
}

/// Информация о коллекции.
struct CollectionInfo: Codable {
    /// Название коллекции
    /// Пример: "Material Design Icons"
    let name: String
}
```

### Пример JSON

```json
{
  "icons": ["mdi:home", "fa:home"],
  "total": 2,
  "collections": {
    "mdi": {
      "name": "Material Design Icons"
    },
    "fa": {
      "name": "Font Awesome"
    }
  }
}
```

## 🎨 IconViewModel

### Назначение

**Presentation модель** — данные для отображения в UI.

### Текущая Реализация (требует доработки)

```swift
import Foundation

/// ViewModel для отображения иконки в ячейке таблицы.
/// 
/// Presentation модель, содержит только данные для UI.
/// Создаётся из Domain модели (Icon) в Presenter.
struct IconViewModel {
    /// Текст размера: "24x24"
    let sizeText: String
    
    /// Текст тегов: "Tags: home, house, building"
    let tagsText: String
    
    /// Полное имя иконки для построения URL
    /// ⚠️ ТРЕБУЕТ ДОРАБОТКИ: лучше хранить URL напрямую
    let iconFullName: String
}
```

### ⚠️ Проблема Текущей Реализации

**Текущее состояние:**
```swift
struct IconViewModel {
    let iconFullName: String  // ← Presenter должен строить URL
}

// В Presenter
private func mapIconsToViewModels(icons: [Icon]) -> [IconViewModel] {
    return icons.map { icon in
        let iconURL = URL(string: "https://api.iconify.design/\(icon.fullName).svg")
        // ⚠️ URL теряется! Не попадает в ViewModel
        return IconViewModel(sizeText: "...", tagsText: "...", iconFullName: icon.fullName)
    }
}
```

**Проблема:** ViewModel не содержит URL картинки, только имя. Это нарушает Separation of Concerns — View должен получать готовые данные.

### ✅ Правильная Реализация

```swift
import Foundation

/// ViewModel для отображения иконки в ячейке таблицы.
/// 
/// Содержит все данные, необходимые View для отображения.
struct IconViewModel {
    /// URL изображения иконки
    /// Пример: https://api.iconify.design/mdi/home.svg
    let iconImageURL: URL?
    
    /// Текст размера: "24x24"
    let sizeText: String
    
    /// Текст тегов: "Tags: home, house, building"
    let tagsText: String
}
```

**В Presenter:**
```swift
private func mapIconsToViewModels(icons: [Icon]) -> [IconViewModel] {
    return icons.map { icon in
        IconViewModel(
            iconImageURL: URL(string: "https://api.iconify.design/\(icon.fullName).svg"),
            sizeText: "\(icon.width)x\(icon.height)",
            tagsText: "Tags: " + icon.tags.prefix(10).joined(separator: ", ")
        )
    }
}
```

## ✅ Best Practices для Моделей

### 1. Разделяй Модели по Слоям

```swift
// ✅ Хорошо
struct Icon { }              // Domain
struct IconViewModel { }     // Presentation
struct IconResponse: Codable { }  // Data

// ❌ Плохо
struct Icon {
    // Всё в одной модели: и данные API, и UI поля
    let fullName: String
    let uiBackgroundColor: UIColor  // ← UI логика в domain!
}
```

### 2. Используй Value Types (Struct)

```swift
// ✅ Хорошо: struct
struct Icon {
    let fullName: String
}

// ❌ Плохо: class без необходимости
class Icon {
    let fullName: String
}
```

**Почему struct:**
- Копируется при присваивании (безопаснее)
- Не нужно думать о памяти
- Нет наследования (проще код)

### 3. Immutable по Умолчанию

```swift
// ✅ Хорошо
struct Icon {
    let fullName: String      // immutable
    var tags: [String] = []   // mutable только если нужно
}

// ❌ Плохо
struct Icon {
    var fullName: String  // Зачем mutable?
}
```

### 4. Computed Properties для Вычислений

```swift
// ✅ Хорошо
var collectionPrefix: String {
    fullName.split(separator: ":").first.map(String.init) ?? ""
}

// ❌ Плохо: храним вычисленное
struct Icon {
    let fullName: String
    let collectionPrefix: String  // Дублирование данных!
}
```

### 5. Codable Только для Data Моделей

```swift
// ✅ Хорошо
struct IconResponse: Codable { }  // Data layer
struct Icon { }                   // Domain layer (не Codable)

// ❌ Плохо
struct Icon: Codable { }  // Domain модель зависит от формата JSON
```

## 🔄 Mapping Between Layers

### Схема Преобразования

```
JSON (API)
    │
    ▼ Decodable
IconDetailResponse (Data)
    │
    ▼ Map to Domain
Icon (Domain)
    │
    ▼ Map to ViewModel
IconViewModel (Presentation)
    │
    ▼ Display
UITableViewCell (View)
```

### Пример в Коде

```swift
// 1. Data → Domain
func parseIcon(from response: IconDetailResponse, prefix: String) -> [Icon] {
    return response.icons.map { name, metadata in
        Icon(
            fullName: "\(prefix):\(name)",
            tags: metadata.tags ?? [],
            width: response.width,
            height: response.height
        )
    }
}

// 2. Domain → ViewModel
func mapToViewModels(icons: [Icon]) -> [IconViewModel] {
    return icons.map { icon in
        IconViewModel(
            iconImageURL: URL(string: "https://api.iconify.design/\(icon.fullName).svg"),
            sizeText: "\(icon.width)x\(icon.height)",
            tagsText: "Tags: " + icon.tags.prefix(10).joined(separator: ", ")
        )
    }
}
```

## 🧪 Тестирование Моделей

### Unit Тест Пример

```swift
import XCTest

final class IconTests: XCTestCase {
    
    func testCollectionPrefix() {
        // Given
        let icon = Icon(fullName: "mdi:home")
        
        // When & Then
        XCTAssertEqual(icon.collectionPrefix, "mdi")
        XCTAssertEqual(icon.iconName, "home")
    }
    
    func testIconNameWithMultipleColons() {
        // Given
        let icon = Icon(fullName: "custom:long:name")
        
        // When & Then
        XCTAssertEqual(icon.collectionPrefix, "custom")
        XCTAssertEqual(icon.iconName, "name")
    }
    
    func testDefaultValues() {
        // Given
        let icon = Icon(fullName: "mdi:home")
        
        // Then
        XCTAssertEqual(icon.tags, [])
        XCTAssertEqual(icon.width, 0)
        XCTAssertEqual(icon.height, 0)
    }
}
```
