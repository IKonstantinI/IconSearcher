## 👁️ IconSearchViewProtocol

### Назначение

**Протокол View** — определяет методы, которые Presenter может вызывать у View.

### Интерфейс

```swift
import UIKit

/// Протокол View для модуля поиска иконок.
/// 
/// Определяет методы для отображения состояний:
/// - Загрузка (loading)
/// - Результаты (icons)
/// - Ошибка (error)
/// - Пустое состояние (empty)
protocol IconSearchViewProtocol: AnyObject {
    
    /// Отобразить список иконок.
    /// - Parameter viewModels: Массив ViewModel для отображения
    func showIcons(viewModels: [IconViewModel])
    
    /// Показать индикатор загрузки.
    func showLoading()
    
    /// Скрыть индикатор загрузки.
    func hideLoading()
    
    /// Показать ошибку.
    /// - Parameters:
    ///   - title: Заголовок ошибки
    ///   - message: Описание ошибки
    func showError(title: String, message: String)
    
    /// Показать пустое состояние (ничего не найдено).
    /// - Parameter query: Поисковый запрос
    func showEmptyState(query: String)
    
    /// Показать начальное состояние (поиск ещё не выполнялся).
    func showInitialState()
}
```

### Реализация в ViewController

```swift
final class IconSearchViewController: UIViewController {
    // ...
}

extension IconSearchViewController: IconSearchViewProtocol {
    
    func showIcons(viewModels: [IconViewModel]) {
        // Сохраняем viewModels
        self.viewModels = viewModels
        
        // Обновляем таблицу
        tableView.reloadData()
        
        // Скрываем empty state, если был
        emptyStateView.isHidden = true
    }
    
    func showLoading() {
        loadingIndicator.startAnimating()
        tableView.isUserInteractionEnabled = false
    }
    
    func hideLoading() {
        loadingIndicator.stopAnimating()
        tableView.isUserInteractionEnabled = true
    }
    
    func showError(title: String, message: String) {
        let alert = UIAlertController(
            title: title,
            message: message,
            preferredStyle: .alert
        )
        alert.addAction(UIAlertAction(title: "OK", style: .default))
        present(alert, animated: true)
    }
    
    func showEmptyState(query: String) {
        emptyStateLabel.text = "Ничего не найдено по запросу \"\(query)\""
        emptyStateView.isHidden = false
        tableView.reloadData()
    }
    
    func showInitialState() {
        viewModels = []
        tableView.reloadData()
        emptyStateView.isHidden = true
    }
}
```

## 🎯 IconSearchPresenterProtocol

### Назначение

**Протокол Presenter** — определяет методы, которые View может вызывать у Presenter.

### Интерфейс

```swift
import Foundation

/// Протокол Presenter для модуля поиска иконок.
/// 
/// Определяет методы для обработки действий пользователя:
/// - Инициализация (viewDidLoad)
/// - Поиск (searchButtonTapped)
/// - Выбор иконки (didSelectIcon)
/// - Сохранение иконки (saveIconToGallery)
protocol IconSearchPresenterProtocol: AnyObject {
    
    /// View загружен и готов к работе.
    func viewDidLoad()
    
    /// Пользователь нажал кнопку поиска.
    /// - Parameter query: Поисковый запрос (может быть nil)
    func searchButtonTapped(query: String?)
    
    /// Пользователь выбрал иконку в таблице.
    /// - Parameter index: Индекс выбранной иконки
    func didSelectIcon(at index: Int)
    
    /// Сохранить иконку в галерею.
    /// - Parameter index: Индекс иконки для сохранения
    func saveIconToGallery(at index: Int)
}
```

### Реализация в Presenter

```swift
final class IconSearchPresenter: IconSearchPresenterProtocol {
    
    private weak var view: IconSearchViewProtocol?
    private let iconService: IconServiceProtocol
    private var icons: [Icon] = []
    
    init(view: IconSearchViewProtocol, iconService: IconServiceProtocol) {
        self.view = view
        self.iconService = iconService
    }
    
    func viewDidLoad() {
        // View готов, можно что-то сделать
        view?.showInitialState()
    }
    
    func searchButtonTapped(query: String?) {
        // 1. Валидация
        guard let query = query, !query.isEmpty else {
            return
        }
        
        // 2. Показываем loading
        view?.showLoading()
        
        // 3. Ищем иконки
        iconService.searchIcons(query: query) { [weak self] result in
            guard let self = self else { return }
            
            self.view?.hideLoading()
            
            switch result {
            case .success(let icons):
                self.icons = icons
                
                if icons.isEmpty {
                    self.view?.showEmptyState(query: query)
                } else {
                    let viewModels = self.mapIconsToViewModels(icons: icons)
                    self.view?.showIcons(viewModels: viewModels)
                }
                
            case .failure(let error):
                self.view?.showError(
                    title: "Ошибка",
                    message: error.localizedDescription
                )
            }
        }
    }
    
    func didSelectIcon(at index: Int) {
        guard icons.indices.contains(index) else { return }
        let selectedIcon = icons[index]
        print("Presenter: User selected icon \(selectedIcon.fullName)")
    }
    
    func saveIconToGallery(at index: Int) {
        guard icons.indices.contains(index) else { return }
        let icon = icons[index]
        
        // Логика сохранения в галерею
        // ...
    }
    
    private func mapIconsToViewModels(icons: [Icon]) -> [IconViewModel] {
        return icons.map { icon in
            IconViewModel(
                iconImageURL: URL(string: "https://api.iconify.design/\(icon.fullName).svg"),
                sizeText: "\(icon.width)x\(icon.height)",
                tagsText: "Tags: " + icon.tags.prefix(10).joined(separator: ", ")
            )
        }
    }
}
```

## 📱 IconCellPresenterDelegate

### Назначение

**Протокол делегата** для связи CellPresenter и TableViewCell.

### Интерфейс

```swift
import UIKit

/// Протокол делегата для IconCellPresenter.
/// 
/// Используется для уведомления ячейки о том, что:
/// - Изображение загружено
/// - Произошла ошибка при загрузке
protocol IconCellPresenterDelegate: AnyObject {
    
    /// Изображение загружено (или взято из кэша).
    /// - Parameters:
    ///   - presenter: Presenter, который загрузил изображение
    ///   - image: Загруженное изображение (nil если ошибка)
    func imageLoader(
        _ presenter: IconCellPresenter,
        didLoad image: UIImage?
    )
    
 /// Произошла ошибка при загрузке изображения.
    /// - Parameters:
    ///   - presenter: Presenter
    ///   - error: Ошибка
    func imageLoader(
        _ presenter: IconCellPresenter,
        didFailWith error: Error
    )
}
```

### Реализация в TableViewCell

```swift
final class IconTableViewCell: UITableViewCell {
    
    @IBOutlet private weak var iconImageView: UIImageView!
    @IBOutlet private weak var sizeLabel: UILabel!
    @IBOutlet private weak var tagsLabel: UILabel!
    
    private var cellPresenter: IconCellPresenter?
    
    override func prepareForReuse() {
        super.prepareForReuse()
        cellPresenter?.cancelLoad()
        iconImageView.image = nil
    }
    
    func configure(
        with viewModel: IconViewModel,
        presenter: IconCellPresenter
    ) {
        self.cellPresenter = presenter
        
        // Текстовые данные
        sizeLabel.text = viewModel.sizeText
        tagsLabel.text = viewModel.tagsText
        
        // Загрузка изображения
        presenter.loadImage(from: viewModel.iconImageURL)
    }
}

extension IconTableViewCell: IconCellPresenterDelegate {
    
    func imageLoader(
        _ presenter: IconCellPresenter,
        didLoad image: UIImage?
    ) {
        // Проверяем, что эта ячейка всё ещё актуальна
        // (не ушла в пул и не переиспользуется)
        iconImageView.image = image
    }
    
    func imageLoader(
        _ presenter: IconCellPresenter,
        didFailWith error: Error
    ) {
        // Показываем placeholder или пустое изображение
        iconImageView.image = UIImage(named: "placeholder")
    }
}
```

## 🎨 IconViewModel

### Назначение

**Presentation модель** — содержит данные для отображения в ячейке.

### Структура

```swift
import Foundation

/// ViewModel для отображения иконки в ячейке таблицы.
/// 
/// Содержит только данные, необходимые View.
/// Создаётся в Presenter из Domain модели (Icon).
struct IconViewModel {
    
    /// URL изображения иконки.
    /// Пример: https://api.iconify.design/mdi/home.svg
    let iconImageURL: URL?
    
    /// Текст размера.
    /// Пример: "24x24"
    let sizeText: String
    
    /// Текст тегов.
    /// Пример: "Tags: home, house, building"
    let tagsText: String
}
```

### Создание в Presenter

```swift
private func mapIconsToViewModels(icons: [Icon]) -> [IconViewModel] {
    return icons.map { icon in
        IconViewModel(
            iconImageURL: URL(string: "https://api.iconify.design/\(icon.fullName).svg"),
            sizeText: "\(icon.width)x\(icon.height)",
            tagsText: "Tags: " + icon.tags.prefix(10).joined(separator: ", ")
        )
    }
}
```

### Использование в View

```swift
// В ViewController
func tableView(_ tableView: UITableView, cellForRowAt indexPath: IndexPath) 
    -> UITableViewCell {
    
    let cell = tableView.dequeueReusableCell(withIdentifier: "IconCell", for: indexPath)
    
    if let iconCell = cell as? IconTableViewCell {
        let viewModel = viewModels[indexPath.row]
        let presenter = cellPresenters[indexPath.row]
        
        iconCell.configure(with: viewModel, presenter: presenter)
    }
    
    return cell
}
```

## ✅ Best Practices

### 1. Weak Reference для View

```swift
// ✅ Хорошо
protocol IconSearchViewProtocol: AnyObject { }

final class IconSearchPresenter {
    private weak var view: IconSearchViewProtocol?
}

// ❌ Плохо
final class IconSearchPresenter {
    private var view: IconSearchViewProtocol  // Сильная ссылка!
}
```

**Почему:** View (ViewController) держит Presenter, Presenter должен держать View слабо.

---

### 2. AnyClass для Протоколов Делегата

```swift
// ✅ Хорошо
protocol IconCellPresenterDelegate: AnyObject { }

// ❌ Плохо
protocol IconCellPresenterDelegate: class { }  // Устаревший синтаксис
```

---

### 3. Говорящие Названия Методов

```swift
// ✅ Хорошо
func showIcons(viewModels: [IconViewModel])
func showLoading()
func showError(title: String, message: String)

// ❌ Плохо
func updateData()
func startLoading()
func handleError()
```

---

### 4. Полные Состояния UI

```swift
// ✅ Хорошо: все состояния
protocol IconSearchViewProtocol {
    func showInitialState()
    func showLoading()
    func showIcons(viewModels: [IconViewModel])
    func showEmptyState(query: String)
    func showError(title: String, message: String)
}

// ❌ Плохо: не все состояния
protocol IconSearchViewProtocol {
    func showIcons(viewModels: [IconViewModel])
    func showLoading()
    func showError(title: String, message: String)
    // Нет showEmptyState и showInitialState
}
```

---

### 5. ViewModel Только для View

```swift
// ✅ Хорошо
struct IconViewModel {
    let iconImageURL: URL?
    let sizeText: String
    let tagsText: String
}

// ❌ Плохо: Domain модель в протоколе View
protocol IconSearchViewProtocol {
    func showIcons(_ icons: [Icon])  // Domain модель!
}
```

## ⚠️ Антипаттерны

### 1. Presenter Знает о ViewController

```swift
// ❌ Плохо
final class IconSearchPresenter {
    private weak var viewController: IconSearchViewController?
}

// ✅ Хорошо
final class IconSearchPresenter {
    private weak var view: IconSearchViewProtocol?
}
```

---

### 2. Нет Weak для Delegate

```swift
// ❌ Плохо
protocol IconCellPresenterDelegate { }

final class IconCellPresenter {
    private var delegate: IconCellPresenterDelegate  // Retain cycle!
}

// ✅ Хорошо
protocol IconCellPresenterDelegate: AnyObject { }

final class IconCellPresenter {
    private weak var delegate: IconCellPresenterDelegate?
}
```

---

### 3. ViewModel с Бизнес-Логикой

```swift
// ❌ Плохо
struct IconViewModel {
    let icon: Icon  // Domain модель внутри ViewModel!
    
    func saveToGallery() {  // Бизнес-логика в ViewModel!
        // ...
    }
}

// ✅ Хорошо
struct IconViewModel {
    let iconImageURL: URL?
    let sizeText: String
    let tagsText: String
    // Только данные, нет логики
}
```
