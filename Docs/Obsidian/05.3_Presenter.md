## 🎯 Назначение Presenter

### Что Решает

```
Проблема: Где размещать бизнес-логику?
├── ❌ В View (ViewController)? → Massive VC
├── ❌ В Service? → Нарушение SRP
└── ✅ В Presenter → Правильно!

Решение: Presenter отвечает за:
- Обработку действий пользователя
- Вызов сервисов (IconService)
- Маппинг Domain → ViewModel
- Управление состояниями View
```

### Зона Ответственности

| Делает | Не Делает |
|--------|-----------|
| ✅ Обработка действий View | ❌ Не знает о UI элементах |
| ✅ Вызов сервисов | ❌ Не отображает данные |
| ✅ Маппинг моделей | ❌ Не работает с UIView |
| ✅ Бизнес-логика | ❌ Не декодирует JSON |

## 💻 IconSearchPresenter

### Реализация

```swift
import Foundation

/// Presenter для модуля поиска иконок.
/// 
/// Отвечает за:
/// - Обработку действий пользователя (поиск, выбор иконки)
/// - Вызов сервисов (IconService)
/// - Маппинг Domain моделей в ViewModel
/// - Управление состояниями View
/// 
/// Архитектура: MVP (Presenter layer)
final class IconSearchPresenter: IconSearchPresenterProtocol {
    
    // MARK: - Properties
    
    /// View для отображения данных
    private weak var view: IconSearchViewProtocol?
    
    /// Сервис поиска иконок
    private let iconService: IconServiceProtocol
    
    /// Список иконок (Domain модели)
    private var icons: [Icon] = []
    
    /// CellPresenter factory (для создания презентеров ячеек)
    private let cellPresenterFactory: IconCellPresenterFactory
    
    // MARK: - Initialization
    
    /// Инициализация с зависимостями.
    /// - Parameters:
    ///   - view: View для отображения (weak reference)
    ///   - iconService: Сервис поиска иконок
    ///   - cellPresenterFactory: Фабрика CellPresenter
    init(
        view: IconSearchViewProtocol,
        iconService: IconServiceProtocol,
        cellPresenterFactory: IconCellPresenterFactory
    ) {
        self.view = view
        self.iconService = iconService
        self.cellPresenterFactory = cellPresenterFactory
    }
    
    // MARK: - IconSearchPresenterProtocol
    
    func viewDidLoad() {
        // View готов, показываем начальное состояние
        view?.showInitialState()
    }
    
    func searchButtonTapped(query: String?) {
        // 1. Валидация запроса
        guard let query = query, !query.isEmpty else {
            return
        }
        
        // 2. Показываем loading
        view?.showLoading()
        
        // 3. Ищем иконки
        iconService.searchIcons(query: query) { [weak self] result in
            guard let self = self else { return }
            
            // 4. Скрываем loading
            self.view?.hideLoading()
            
            // 5. Обрабатываем результат
            switch result {
            case .success(let icons):
                self.handleSuccess(icons: icons, query: query)
                
            case .failure(let error):
                self.handleFailure(error: error)
            }
        }
    }
    
    func didSelectIcon(at index: Int) {
        guard icons.indices.contains(index) else { return }
        let selectedIcon = icons[index]
        
        // Логирование выбора (можно отправить аналитику)
        print("Presenter: User selected icon \(selectedIcon.fullName)")
        
        // Сохранение в галерею
        saveIconToGallery(at: index)
    }
    
    func saveIconToGallery(at index: Int) {
        guard icons.indices.contains(index) else { return }
        let icon = icons[index]
        
        // Логика сохранения в галерею
        // Загрузка изображения и сохранение через UIImageWriteToSavedPhotosAlbum
        // ...
    }
    
    // MARK: - Private Methods
    
    private func handleSuccess(icons: [Icon], query: String) {
        self.icons = icons
        
        if icons.isEmpty {
            view?.showEmptyState(query: query)
        } else {
            let viewModels = mapIconsToViewModels(icons: icons)
            view?.showIcons(viewModels: viewModels)
        }
    }
    
    private func handleFailure(error: Error) {
        view?.showError(
            title: "Ошибка",
            message: error.localizedDescription
        )
    }
    
    /// Маппинг Domain моделей в ViewModel.
    /// - Parameter icons: Domain модели
    /// - Returns: Presentation модели для View
    private func mapIconsToViewModels(icons: [Icon]) -> [IconViewModel] {
        return icons.map { icon in
            IconViewModel(
                iconImageURL: URL(string: "https://api.iconify.design/\(icon.fullName).svg"),
                sizeText: "\(icon.width)x\(icon.height)",
                tagsText: "Tags: " + icon.tags.prefix(10).joined(separator: ", ")
            )
        }
    }
}
```

## 🔄 Маппинг Моделей

### Схема Преобразования

```
┌─────────────────────────────────────────────────────────────┐
│  Domain Layer → Presentation Layer                          │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│  Icon (Domain)              IconViewModel (Presentation)    │
│  ┌─────────────────┐        ┌─────────────────────────┐    │
│  │ fullName: String│        │ iconImageURL: URL?      │    │
│  │ tags: [String]  │ ─────▶ │ sizeText: String        │    │
│  │ width: Int      │        │ tagsText: String        │    │
│  │ height: Int     │        └─────────────────────────┘    │
│  └─────────────────┘                                        │
│                                                             │
│  Пример:                                                    │
│  Icon(                        IconViewModel(                │
│    fullName: "mdi:home",        iconImageURL: URL(...),     │
│    tags: ["house", "home"],     sizeText: "24x24",          │
│    width: 24,                   tagsText: "Tags: house..."  │
│    height: 24                                               │
│  )                                                          │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

### Почему Маппинг Важен

**Без маппинга:**
```swift
// ❌ Плохо: View знает про Domain модель
protocol IconSearchViewProtocol {
    func showIcons(_ icons: [Icon])  // Domain модель!
}

// View должен импортировать Domain layer
// Нарушение Separation of Concerns
```

**С маппингом:**
```swift
// ✅ Хорошо: View знает только ViewModel
protocol IconSearchViewProtocol {
    func showIcons(viewModels: [IconViewModel])  // Presentation модель!
}

// View импортирует только Presentation layer
// Правильное разделение ответственности
```

## ⚠️ Обработка Ошибок

### Типы Ошибок

```swift
// 1. Ошибки валидации
enum ValidationError: Error {
    case emptyQuery
    case queryTooShort(minLength: Int)
}

// 2. Ошибки сервиса (из IconService)
enum ServiceError: Error {
    case networkError(Error)
    case parsingError(Error)
    case noResults
}

// 3. Ошибки UI
enum UIError: Error {
    case cannotDisplay
    case cannotSaveToGallery
}
```

### Стратегия Обработки

```swift
final class IconSearchPresenter {
    
    func searchButtonTapped(query: String?) {
        // 1. Валидация
        guard let query = query, !query.isEmpty else {
            // Не показываем ошибку, просто игнорируем
            return
        }
        
        // 2. Дополнительная валидация
        if query.count < 3 {
            view?.showError(
                title: "Ошибка",
                message: "Запрос должен быть не менее 3 символов"
            )
            return
        }
        
        // 3. Показываем loading
        view?.showLoading()
        
        // 4. Вызов сервиса
        iconService.searchIcons(query: query) { [weak self] result in
            guard let self = self else { return }
            
            self.view?.hideLoading()
            
            switch result {
            case .success(let icons):
                if icons.isEmpty {
                    // Пустой результат — не ошибка!
                    self.view?.showEmptyState(query: query)
                } else {
                    let viewModels = self.mapIconsToViewModels(icons: icons)
                    self.view?.showIcons(viewModels: viewModels)
                }
                
            case .failure(let error):
                // Ошибка сервиса
                self.handleServiceError(error)
            }
        }
    }
    
    private func handleServiceError(_ error: Error) {
        // Логирование ошибки (можно отправить в аналитику)
        print("Service error: \(error.localizedDescription)")
        
        // Показываем пользователю
        view?.showError(
            title: "Ошибка",
            message: error.localizedDescription
        )
    }
}
```

## ✅ Best Practices

### 1. Weak Reference на View

```swift
// ✅ Хорошо
final class IconSearchPresenter {
    private weak var view: IconSearchViewProtocol?
}

// ❌ Плохо
final class IconSearchPresenter {
    private var view: IconSearchViewProtocol  // Retain cycle!
}
```

**Почему:** ViewController → Presenter (сильная), Presenter → View (слабая).

---

### 2. Протокол для Сервиса

```swift
// ✅ Хорошо
final class IconSearchPresenter {
    private let iconService: IconServiceProtocol
}

// ❌ Плохо
final class IconSearchPresenter {
    private let iconService: IconifyService  // Concrete implementation
}
```

---

### 3. Маппинг в Отдельном Методе

```swift
// ✅ Хорошо
private func mapIconsToViewModels(icons: [Icon]) -> [IconViewModel] {
    return icons.map { icon in
        IconViewModel(
            iconImageURL: URL(string: "https://api.iconify.design/\(icon.fullName).svg"),
            sizeText: "\(icon.width)x\(icon.height)",
            tagsText: "Tags: " + icon.tags.prefix(10).joined(separator: ", ")
        )
    }
}

// ❌ Плохо: маппинг внутри closure
iconService.searchIcons { [weak self] result in
    switch result {
    case .success(let icons):
        let viewModels = icons.map { ... }  // Логика в closure!
        self?.view?.showIcons(viewModels: viewModels)
    }
}
```

---

### 4. Обработка Пустого Результата

```swift
// ✅ Хорошо
switch result {
case .success(let icons):
    if icons.isEmpty {
        view?.showEmptyState(query: query)  // Не ошибка!
    } else {
        // Показываем результаты
    }
}

// ❌ Плохо
switch result {
case .success(let icons):
    // [] тоже показываем как результаты
    let viewModels = self.mapIconsToViewModels(icons: icons)
    view?.showIcons(viewModels: viewModels)
}
```

## ⚠️ Антипаттерны

### 1. Massive Presenter

```swift
// ❌ Плохо: Presenter делает слишком много
final class IconSearchPresenter {
    // Сеть напрямую
    func loadData() {
        URLSession.shared.dataTask(with: url) { ... }
    }
    
    // Кэш напрямую
    func cacheImage(_ image: UIImage) {
        cache.setValue(image, forKey: url)
    }
    
    // Форматирование
    func formatDate(_ date: Date) -> String {
        // ...
    }
    
    // 500+ строк кода
}

// ✅ Хорошо: только бизнес-логика
final class IconSearchPresenter {
    private let iconService: IconServiceProtocol
    
    func searchButtonTapped(query: String?) {
        iconService.searchIcons(query: query) { ... }
    }
}
```

---

### 2. View Знает о Сервисе

```swift
// ❌ Плохо
final class IconSearchViewController: UIViewController {
    private let iconService = IconifyService()
    
    @IBAction func searchButtonTapped() {
        iconService.searchIcons { ... }  // Сервис в View!
    }
}

// ✅ Хорошо
final class IconSearchViewController: UIViewController {
    private let presenter: IconSearchPresenterProtocol
    
    @IBAction func searchButtonTapped() {
        presenter.searchButtonTapped(query: textField.text)
    }
}
```

---

### 3. Domain Модель в View

```swift
// ❌ Плохо
protocol IconSearchViewProtocol {
    func showIcons(_ icons: [Icon])  // Domain!
}

// ✅ Хорошо
protocol IconSearchViewProtocol {
    func showIcons(viewModels: [IconViewModel])  // Presentation!
}
```

---

### 4. Нет Weak Self в Closure

```swift
// ❌ Плохо
iconService.searchIcons { [weak self] result in
    self?.handleSuccess(icons: icons)  // ✅ Хорошо
}

iconService.searchIcons { result in
    self.handleSuccess(icons: icons)  // ❌ Retain cycle!
}
```

## 🧪 Тестирование

### Mock для Тестов

```swift
import Foundation

/// Mock View для тестов Presenter.
final class MockIconSearchView: IconSearchViewProtocol {
    
    var showIconsCalled = false
    var showLoadingCalled = false
    var hideLoadingCalled = false
    var showErrorCalled = false
    var showEmptyStateCalled = false
    var showInitialStateCalled = false
    
    var lastViewModels: [IconViewModel]?
    var lastErrorTitle: String?
    var lastErrorMessage: String?
    var lastEmptyQuery: String?
    
    func showIcons(viewModels: [IconViewModel]) {
        showIconsCalled = true
        lastViewModels = viewModels
    }
    
    func showLoading() {
        showLoadingCalled = true
    }
    
    func hideLoading() {
        hideLoadingCalled = true
    }
    
    func showError(title: String, message: String) {
        showErrorCalled = true
        lastErrorTitle = title
        lastErrorMessage = message
    }
    
    func showEmptyState(query: String) {
        showEmptyStateCalled = true
        lastEmptyQuery = query
    }
    
    func showInitialState() {
        showInitialStateCalled = true
    }
}
```

### Unit Тест Пример

```swift
import XCTest

final class IconSearchPresenterTests: XCTestCase {
    
    var presenter: IconSearchPresenter!
    var mockView: MockIconSearchView!
    var mockService: MockIconService!
    
    override func setUp() {
        super.setUp()
        mockView = MockIconSearchView()
        mockService = MockIconService()
        presenter = IconSearchPresenter(
            view: mockView,
            iconService: mockService,
            cellPresenterFactory: MockCellPresenterFactory()
        )
    }
    
    func testSearchButtonTapped_Success() {
        // Given
        mockService.mockResult = .success([
            Icon(fullName: "mdi:home")
        ])
        
        // When
        presenter.searchButtonTapped(query: "home")
        
        // Then
        XCTAssertTrue(mockView.showLoadingCalled)
        XCTAssertTrue(mockView.hideLoadingCalled)
        XCTAssertTrue(mockView.showIconsCalled)
        XCTAssertFalse(mockView.showEmptyStateCalled)
    }
    
    func testSearchButtonTapped_EmptyResult() {
        // Given
        mockService.mockResult = .success([])
        
        // When
        presenter.searchButtonTapped(query: "xyz123")
        
        // Then
        XCTAssertTrue(mockView.hideLoadingCalled)
        XCTAssertTrue(mockView.showEmptyStateCalled)
        XCTAssertEqual(mockView.lastEmptyQuery, "xyz123")
    }
    
    func testSearchButtonTapped_Error() {
        // Given
        mockService.mockResult = .failure(URLError(.notConnectedToInternet))
        
        // When
        presenter.searchButtonTapped(query: "home")
        
        // Then
        XCTAssertTrue(mockView.hideLoadingCalled)
        XCTAssertTrue(mockView.showErrorCalled)
        XCTAssertEqual(mockView.lastErrorTitle, "Ошибка")
    }
}
```
