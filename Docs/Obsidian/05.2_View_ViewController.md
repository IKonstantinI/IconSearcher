## 🎯 Назначение

### Что Решает

```
Проблема: Где размещать UI логику?
├── ❌ В Presenter? → Нарушение Separation of Concerns
├── ❌ В Service? → Нарушение Single Responsibility
└── ✅ В View (ViewController) → Правильно!

Решение: IconSearchViewController отвечает за:
- Отображение UI элементов
- Обработку касаний
- Вызов методов Presenter
- Реализацию протокола IconSearchViewProtocol
```

### Зона Ответственности

| Делает | Не Делает |
|--------|-----------|
| ✅ Отображение UI | ❌ Не содержит бизнес-логики |
| ✅ Обработка касаний | ❌ Не знает про API |
| ✅ Вызов Presenter | ❌ Не создаёт ViewModel |
| ✅ Анимации UI | ❌ Не декодирует JSON |

## 💻 Реализация

### Полная Версия

```swift
import UIKit

/// ViewController для модуля поиска иконок.
/// 
/// Отвечает за:
/// - Отображение UI (textField, button, tableView)
/// - Обработку действий пользователя
/// - Вызов методов Presenter
/// 
/// Архитектура: MVP (View layer)
final class IconSearchViewController: UIViewController {
    
    // MARK: - UI Elements
    
    private let searchTextField: UITextField = {
        let textField = UITextField()
        textField.placeholder = "Поиск иконок..."
        textField.borderStyle = .roundedRect
        textField.returnKeyType = .search
        textField.autocapitalizationType = .none
        textField.autocorrectionType = .no
        return textField
    }()
    
    private let searchButton: UIButton = {
        let button = UIButton(type: .system)
        button.setTitle("Поиск", for: .normal)
        button.titleLabel?.font = UIFont.systemFont(ofSize: 16, weight: .medium)
        return button
    }()
    
    private let tableView: UITableView = {
        let tableView = UITableView(frame: .zero, style: .plain)
        tableView.register(
            IconTableViewCell.self,
            forCellReuseIdentifier: "IconCell"
        )
        tableView.tableFooterView = UIView()  // Убираем пустые ячейки
        return tableView
    }()
    
    private let loadingIndicator: UIActivityIndicatorView = {
        let indicator = UIActivityIndicatorView(style: .large)
        indicator.hidesWhenStopped = true
        return indicator
    }()
    
    private let emptyStateView: UIView = {
        let view = UIView()
        view.backgroundColor = .systemBackground
        view.isHidden = true
        return view
    }()
    
    private let emptyStateLabel: UILabel = {
        let label = UILabel()
        label.textAlignment = .center
        label.textColor = .secondaryLabel
        label.font = UIFont.systemFont(ofSize: 18, weight: .regular)
        return label
    }()
    
    // MARK: - Properties
    
    /// ViewModel для отображения
    private var viewModels: [IconViewModel] = []
    
    /// Presenter для каждой ячейки (создаётся динамически)
    private var cellPresenters: [IconCellPresenter] = []
    
    /// Presenter модуля
    private let presenter: IconSearchPresenterProtocol
    
    // MARK: - Initialization
    
    init(presenter: IconSearchPresenterProtocol) {
        self.presenter = presenter
        super.init(nibName: nil, bundle: nil)
    }
    
    required init?(coder: NSCoder) {
        fatalError("init(coder:) has not been implemented")
    }
    
    // MARK: - Lifecycle
    
    override func viewDidLoad() {
        super.viewDidLoad()
        setupUI()
        setupConstraints()
        setupDelegates()
        presenter.viewDidLoad()
    }
    
    // MARK: - UI Setup
    
    private func setupUI() {
        view.backgroundColor = .systemBackground
        title = "Поиск иконок"
        
        // Добавляем элементы
        view.addSubview(searchTextField)
        view.addSubview(searchButton)
        view.addSubview(tableView)
        view.addSubview(loadingIndicator)
        view.addSubview(emptyStateView)
        emptyStateView.addSubview(emptyStateLabel)
        
        // Центрируем loading
        loadingIndicator.center = view.center
    }
    
    private func setupConstraints() {
        searchTextField.translatesAutoresizingMaskIntoConstraints = false
        searchButton.translatesAutoresizingMaskIntoConstraints = false
        tableView.translatesAutoresizingMaskIntoConstraints = false
        emptyStateView.translatesAutoresizingMaskIntoConstraints = false
        emptyStateLabel.translatesAutoresizingMaskIntoConstraints = false
        
        NSLayoutConstraint.activate([
            // Search TextField
            searchTextField.topAnchor.constraint(equalTo: view.safeAreaLayoutGuide.topAnchor, constant: 16),
            searchTextField.leadingAnchor.constraint(equalTo: view.leadingAnchor, constant: 16),
            searchTextField.trailingAnchor.constraint(equalTo: searchButton.leadingAnchor, constant: -8),
            searchTextField.heightAnchor.constraint(equalToConstant: 44),
            
            // Search Button
            searchButton.topAnchor.constraint(equalTo: view.safeAreaLayoutGuide.topAnchor, constant: 16),
            searchButton.trailingAnchor.constraint(equalTo: view.trailingAnchor, constant: -16),
            searchButton.widthAnchor.constraint(equalToConstant: 80),
            searchButton.heightAnchor.constraint(equalToConstant: 44),
            
            // TableView
            tableView.topAnchor.constraint(equalTo: searchTextField.bottomAnchor, constant: 16),
            tableView.leadingAnchor.constraint(equalTo: view.leadingAnchor),
            tableView.trailingAnchor.constraint(equalTo: view.trailingAnchor),
            tableView.bottomAnchor.constraint(equalTo: view.bottomAnchor),
            
            // Empty State View
            emptyStateView.topAnchor.constraint(equalTo: tableView.topAnchor),
            emptyStateView.leadingAnchor.constraint(equalTo: tableView.leadingAnchor),
            emptyStateView.trailingAnchor.constraint(equalTo: tableView.trailingAnchor),
            emptyStateView.bottomAnchor.constraint(equalTo: tableView.bottomAnchor),
            
            // Empty State Label
            emptyStateLabel.centerXAnchor.constraint(equalTo: emptyStateView.centerXAnchor),
            emptyStateLabel.centerYAnchor.constraint(equalTo: emptyStateView.centerYAnchor)
        ])
    }
    
    private func setupDelegates() {
        tableView.dataSource = self
        tableView.delegate = self
        searchTextField.delegate = self
        searchButton.addTarget(self, action: #selector(searchButtonTapped), for: .touchUpInside)
    }
    
    // MARK: - Actions
    
    @objc private func searchButtonTapped() {
        searchTextField.resignFirstResponder()
        presenter.searchButtonTapped(query: searchTextField.text)
    }
}
```

## 🎭 Реализация Протокола

### IconSearchViewProtocol

```swift
// MARK: - IconSearchViewProtocol

extension IconSearchViewController: IconSearchViewProtocol {
    
    func showIcons(viewModels: [IconViewModel]) {
        self.viewModels = viewModels
        tableView.reloadData()
        emptyStateView.isHidden = true
    }
    
    func showLoading() {
        loadingIndicator.startAnimating()
        tableView.isUserInteractionEnabled = false
        emptyStateView.isHidden = true
    }
    
    func hideLoading() {
        loadingIndicator.stopAnimating()
        tableView.isUserInteractionEnabled = true
    }
    
    func showError(title: String, message: String) {
        let alert = UIAlertController(
            title: title,
            message: message,
            preferredStyle: .alert
        )
        alert.addAction(UIAlertAction(title: "OK", style: .default))
        present(alert, animated: true)
    }
    
    func showEmptyState(query: String) {
        emptyStateLabel.text = "Ничего не найдено по запросу \"\(query)\""
        emptyStateView.isHidden = false
        tableView.reloadData()
    }
    
    func showInitialState() {
        viewModels = []
        tableView.reloadData()
        emptyStateView.isHidden = true
    }
}
```

## 📋 UITableViewDataSource & Delegate

```swift
// MARK: - UITableViewDataSource

extension IconSearchViewController: UITableViewDataSource {
    
    func tableView(_ tableView: UITableView, numberOfRowsInSection section: Int) -> Int {
        return viewModels.count
    }
    
    func tableView(_ tableView: UITableView, 
                   cellForRowAt indexPath: IndexPath) -> UITableViewCell {
        
        guard let cell = tableView.dequeueReusableCell(
            withIdentifier: "IconCell",
            for: indexPath
        ) as? IconTableViewCell else {
            return UITableViewCell()
        }
        
        let viewModel = viewModels[indexPath.row]
        
        // Создаём CellPresenter для этой ячейки
        let cellPresenter = IconCellPresenter(
            delegate: cell,
            imageLoader: ImageLoader()  // Или через Assembly
        )
        
        // Сохранем presenter (можно использовать слабый массив)
        if cellPresenters.count <= indexPath.row {
            cellPresenters.append(cellPresenter)
        } else {
            cellPresenters[indexPath.row] = cellPresenter
        }
        
        cell.configure(with: viewModel, presenter: cellPresenter)
        
        return cell
    }
}

// MARK: - UITableViewDelegate

extension IconSearchViewController: UITableViewDelegate {
    
    func tableView(_ tableView: UITableView, 
                   heightForRowAt indexPath: IndexPath) -> CGFloat {
        return 80  // Фиксированная высота
    }
    
    func tableView(_ tableView: UITableView, 
                   didSelectRowAt indexPath: IndexPath) {
        tableView.deselectRow(at: indexPath, animated: true)
        presenter.didSelectIcon(at: indexPath.row)
    }
    
    func tableView(_ tableView: UITableView, 
                   trailingSwipeActionsConfigurationForRowAt indexPath: IndexPath) 
        -> UISwipeActionsConfiguration? {
        
        let saveAction = UIContextualAction(
            style: .normal,
            title: "Сохранить"
        ) { [weak self] _, _, completion in
            self?.presenter.saveIconToGallery(at: indexPath.row)
            completion(true)
        }
        saveAction.backgroundColor = .systemGreen
        
        return UISwipeActionsConfiguration(actions: [saveAction])
    }
}
```

### Схема Состояний

```
┌─────────────────────────────────────────────────────────────┐
│                    UI States                                │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│  ┌─────────────┐                                            │
│  │  Initial    │ ──> Пользователь ещё не искал             │
│  └──────┬──────┘                                            │
│         │ searchButtonTapped                                │
│         ▼                                                   │
│  ┌─────────────┐                                            │
│  │  Loading    │ ──> Показываем indicator                  │
│  └──────┬──────┘                                            │
│         │                                                   │
│    ┌────┴────┐                                              │
│    │         │                                              │
│    ▼         ▼                                              │
│ ┌──────┐ ┌──────────┐                                       │
│ │Empty │ │ Results  │                                       │
│ │State │ │ (Icons)  │                                       │
│ └──────┘ └──────────┘                                       │
│                                                             │
│  В любом состоянии может перейти в:                         │
│  ┌─────────────┐                                            │
│  │   Error     │ ──> Alert с ошибкой                       │
│  └─────────────┘                                            │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

### Реализация Состояний

```swift
enum UIState {
    case initial
    case loading
    case results([IconViewModel])
    case empty(query: String)
    case error(title: String, message: String)
}

final class IconSearchViewController: UIViewController {
    
    private var currentState: UIState = .initial {
        didSet {
            render(state: currentState)
        }
    }
    
    private func render(state: UIState) {
        switch state {
        case .initial:
            showInitialState()
        case .loading:
            showLoading()
        case .results(let viewModels):
            showIcons(viewModels: viewModels)
        case .empty(let query):
            showEmptyState(query: query)
        case .error(let title, let message):
            showError(title: title, message: message)
        }
    }
}
```

## ✅ Best Practices

### 1. Programmatically UI

```swift
// ✅ Хорошо: программно
private let searchTextField: UITextField = {
    let textField = UITextField()
    textField.placeholder = "Поиск иконок..."
    textField.borderStyle = .roundedRect
    return textField
}()

// ❌ Плохо: Storyboard
// - Сложнее version control
// - Меньше контроля
// - Merge conflicts
```

### 2. Dependency Injection

```swift
// ✅ Хорошо
init(presenter: IconSearchPresenterProtocol) {
    self.presenter = presenter
    super.init(nibName: nil, bundle: nil)
}

// ❌ Плохо
init() {
    self.presenter = IconSearchPresenter()  // Создание внутри!
    super.init(nibName: nil, bundle: nil)
}
```

### 3. Weak Self в Closure

```swift
// ✅ Хорошо
searchButton.addTarget(self, action: #selector(searchButtonTapped), for: .touchUpInside)

// В closure
someAsyncMethod { [weak self] result in
    self?.updateUI()
}

// ❌ Плохо
someAsyncMethod {
    self.updateUI()  // Retain cycle!
}
```

### 4. Main Thread для UI

```swift
// ✅ Хорошо (уже в main, это UI)
func showIcons(viewModels: [IconViewModel]) {
    self.viewModels = viewModels
    tableView.reloadData()
}

// ❌ Плохо
DispatchQueue.global().async {
    self.tableView.reloadData()  // Crash!
}
```

## ⚠️ Антипаттерны

### 1. Бизнес-Логика в View

```swift
// ❌ Плохо
func searchButtonTapped() {
    guard let query = searchTextField.text, !query.isEmpty else {
        return
    }
    
    // Бизнес-логика в View!
    if query.count < 3 {
        showError(title: "Error", message: "Query too short")
        return
    }
    
    // Сеть в View!
    networkManager.request(url: url) { ... }
}

// ✅ Хорошо
func searchButtonTapped() {
    // Только делегирование
    presenter.searchButtonTapped(query: searchTextField.text)
}
```

### 2. Massive ViewController

```swift
// ❌ Плохо: 500+ строк
final class IconSearchViewController: UIViewController {
    // Всё в одном классе
    // Сеть, кэш, UI, бизнес-логика
}

// ✅ Хорошо: разделение
final class IconSearchViewController: UIViewController {
    // Только UI
    // Бизнес-логика в Presenter
}
```

### 3. Нет prepareForReuse в Ячейке

```swift
// ❌ Плохо: в ViewController
func tableView(_ tableView: UITableView, cellForRowAt indexPath: IndexPath) {
    let cell = tableView.dequeueReusableCell(...)
    cell.imageView.image = loadImage(...)  // Загрузка в VC!
    return cell
}

// ✅ Хорошо: в Ячейке
func tableView(_ tableView: UITableView, cellForRowAt indexPath: IndexPath) {
    let cell = tableView.dequeueReusableCell(...) as? IconTableViewCell
    cell.configure(with: viewModel, presenter: cellPresenter)
    return cell
}
```
