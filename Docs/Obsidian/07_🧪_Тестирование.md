## 🧪 Unit Тесты

### Что Тестировать

| Компонент | Что Тестировать |
|-----------|-----------------|
| **Cache** | Запись, чтение, удаление, очистка |
| **Icon** | Computed properties (collectionPrefix, iconName) |
| **IconSearchPresenter** | Поиск, маппинг, обработка ошибок |
| **IconCellPresenter** | Загрузка, отмена, delegate |
| **IconifyService** | Парсинг ответов API |

### Структура Теста: AAA Pattern

```swift
func testSearchButtonTapped_Success() {
    // ┌─ Arrange (Подготовка)
    // │  - Создаём mock объекты
    // │  - Настраиваем данные
    // ▼
    let mockView = MockIconSearchView()
    let mockService = MockIconService()
    mockService.mockResult = .success([Icon(fullName: "mdi:home")])
    
    let presenter = IconSearchPresenter(
        view: mockView,
        iconService: mockService
    )
    
    // ┌─ Act (Действие)
    // │  - Вызываем тестируемый метод
    // ▼
    presenter.searchButtonTapped(query: "home")
    
    // ┌─ Assert (Проверка)
    // │  - Проверяем результат
    // │  - Используем XCTAssertEqual, XCTAssertTrue, etc.
    // ▼
    XCTAssertTrue(mockView.showLoadingCalled)
    XCTAssertTrue(mockView.hideLoadingCalled)
    XCTAssertTrue(mockView.showIconsCalled)
}
```

## 📱 UI Тесты

### Что Тестировать

| Сценарий | Что Проверяем |
|----------|---------------|
| **Поиск иконок** | Ввод запроса → нажатие кнопки → результаты |
| **Пустой результат** | Ввод "xyz123" → empty state |
| **Ошибка сети** | Отключение сети → error alert |
| **Выбор иконки** | Тап по ячейке → сохранение в галерею |

### Структура UI Теста

```swift
func testSearchIcons_Success() {
    // 1. Запускаем приложение
    let app = XCUIApplication()
    app.launch()
    
    // 2. Находим элементы
    let searchField = app.textFields["searchTextField"]
    let searchButton = app.buttons["searchButton"]
    let tableView = app.tables["iconTableView"]
    
    // 3. Вводим запрос
    searchField.tap()
    searchField.typeText("home")
    
    // 4. Нажимаем кнопку
    searchButton.tap()
    
    // 5. Ждём появления результатов
    let predicate = NSPredicate(format: "exists == true")
    let expectation = XCTNSPredicateExpectation(
        predicate: predicate,
        object: tableView.cells.element(boundBy: 0)
    )
    XCTWaiter().wait(for: [expectation], timeout: 5.0)
    
    // 6. Проверяем результат
    XCTAssertTrue(tableView.cells.element(boundBy: 0).exists)
}
```

## 🎭 Mock Объекты

### Mock IconService

```swift
import Foundation

/// Mock IconService для unit тестов.
final class MockIconService: IconServiceProtocol {
    
    // Флаги вызова методов
    var searchIconsCalled = false
    var searchIconsCallCount = 0
    
    // Последний запрошенный query
    var lastQuery: String?
    
    // Mock результат
    var mockResult: Result<[Icon], Error>?
    
    // Задержка для асинхронных тестов
    var delay: TimeInterval = 0
    
    func searchIcons(
        query: String,
        completion: @escaping (Result<[Icon], Error>) -> Void
    ) {
        searchIconsCalled = true
        searchIconsCallCount += 1
        lastQuery = query
        
        // Имитация задержки сети
        DispatchQueue.main.asyncAfter(deadline: .now() + delay) {
            if let result = self.mockResult {
                completion(result)
            } else {
                completion(.success([]))
            }
        }
    }
}
```

### Mock ImageLoader

```swift
import UIKit

/// Mock ImageLoader для тестов.
final class MockImageLoader: ImageLoaderProtocol {
    
    var loadImageCalled = false
    var cancelLoadCalled = false
    var clearCacheCalled = false
    
    var lastURL: URL?
    var lastTaskId: Int = -1
    
    var mockImage: UIImage?
    var delay: TimeInterval = 0
    
    func loadImage(
        from url: URL,
        completion: @escaping (UIImage?) -> Void
    ) -> Int {
        loadImageCalled = true
        lastURL = url
        
        DispatchQueue.main.asyncAfter(deadline: .now() + delay) {
            completion(self.mockImage)
        }
        
        return 1
    }
    
    func cancelLoad(taskId: Int) {
        cancelLoadCalled = true
        lastTaskId = taskId
    }
    
    func clearCache() {
        clearCacheCalled = true
    }
}
```

## ✅ Best Practices

### 1. Названия Тестов

```swift
// ✅ Хорошо: понятно, что тестируется
func testSearchButtonTapped_WithValidQuery_ShowsLoading()
func testSearchButtonTapped_WithEmptyQuery_DoesNothing()
func testSearchButtonTapped_WithNetworkError_ShowsError()
func testSearchButtonTapped_WithEmptyResult_ShowsEmptyState()

// ❌ Плохо: непонятно
func testSearch1()
func testSearchSuccess()
func testError()
```

### 2. Один Тест — Одна Концепция

```swift
// ✅ Хорошо
func testSearchButtonTapped_ShowsLoading() {
    // Тестируем только showLoading
}

func testSearchButtonTapped_HidesLoading() {
    // Тестируем только hideLoading
}

func testSearchButtonTapped_ShowsIcons() {
    // Тестируем только showIcons
}

// ❌ Плохо: всё в одном
func testSearchButtonTapped() {
    // Тестируем loading, hideLoading, showIcons, ошибки...
    // Сложно понять, что сломалось
}
```

### 3. Не Зависеть от Реальной Сети

```swift
// ❌ Плохо
func testSearchIcons_RealAPI() {
    let service = IconifyService()  // Реальный сервис!
    // Тест упадёт без сети или при изменении API
}

// ✅ Хорошо
func testSearchIcons_MockService() {
    let mockService = MockIconService()
    mockService.mockResult = .success([...])
    // Тест работает без сети, быстрый, стабильный
}
```

### 4. Тестировать Граничные Случаи

```swift
// Граничные случаи для поиска
func testSearch_WithEmptyQuery()
func testSearch_WithSingleCharacter()
func testSearch_WithVeryLongQuery()
func testSearch_WithSpecialCharacters()
func testSearch_WithWhitespace()

// Граничные случаи для результатов
func testSearch_WithNoResults()
func testSearch_WithOneResult()
func testSearch_WithManyResults()
```

## 📝 Примеры Тестов

### Тест Cache

```swift
import XCTest

final class CacheTests: XCTestCase {
    
    var cache: Cache<String, UIImage>!
    
    override func setUp() {
        super.setUp()
        cache = Cache()
        cache.countLimit = 10
    }
    
    override func tearDown() {
        cache = nil
        super.tearDown()
    }
    
    func testSetValueAndGetValue() {
        // Arrange
        let image = UIImage()
        let key = "test_key"
        
        // Act
        cache.setValue(image, forKey: key)
        let retrieved = cache.value(forKey: key)
        
        // Assert
        XCTAssertNotNil(retrieved)
        XCTAssertEqual(retrieved, image)
    }
    
    func testGetValueForNonExistentKey() {
        // Act
        let retrieved = cache.value(forKey: "non_existent")
        
        // Assert
        XCTAssertNil(retrieved)
    }
    
    func testRemoveValue() {
        // Arrange
        cache.setValue(UIImage(), forKey: "key")
        
        // Act
        cache.removeValue(forKey: "key")
        
        // Assert
        XCTAssertNil(cache.value(forKey: "key"))
    }
    
    func testRemoveAll() {
        // Arrange
        cache.setValue(UIImage(), forKey: "key1")
        cache.setValue(UIImage(), forKey: "key2")
        
        // Act
        cache.removeAll()
        
        // Assert
        XCTAssertNil(cache.value(forKey: "key1"))
        XCTAssertNil(cache.value(forKey: "key2"))
    }
}
```

### Тест IconSearchPresenter

```swift
import XCTest

final class IconSearchPresenterTests: XCTestCase {
    
    var presenter: IconSearchPresenter!
    var mockView: MockIconSearchView!
    var mockService: MockIconService!
    
    override func setUp() {
        super.setUp()
        mockView = MockIconSearchView()
        mockService = MockIconService()
        presenter = IconSearchPresenter(
            view: mockView,
            iconService: mockService,
            imageLoader: MockImageLoader()
        )
    }
    
    func testViewDidLoad_ShowsInitialState() {
        // Act
        presenter.viewDidLoad()
        
        // Assert
        XCTAssertTrue(mockView.showInitialStateCalled)
    }
    
    func testSearchButtonTapped_WithValidQuery() {
        // Arrange
        mockService.mockResult = .success([
            Icon(fullName: "mdi:home")
        ])
        
        // Act
        presenter.searchButtonTapped(query: "home")
        
        // Assert
        XCTAssertTrue(mockView.showLoadingCalled)
        XCTAssertTrue(mockView.hideLoadingCalled)
        XCTAssertTrue(mockView.showIconsCalled)
        XCTAssertEqual(mockView.lastViewModels?.count, 1)
    }

    func testSearchButtonTapped_WithNilQuery() {
        // Act

        XCTAssertFalse(mockView.showLoadingCalled)
        XCTAssertFalse(mockService.searchIconsCalled)

    func testSearchButtonTapped_WithEmptyResult() {
```
