## 🚨 Критические Ошибки

### 1. Нет Debouncer для Поиска

**Проблема:** Запрос на сервер при вводе каждой буквы.

```swift
// ❌ Плохо: запрос на каждый символ
func textField(_ textField: UITextField, 
               shouldChangeCharactersIn range: NSRange, 
               replacementString string: String) -> Bool {
    presenter.searchButtonTapped(query: textField.text)
    return true
}

// Результат:
// Ввод "icon" → 4 запроса к API
// Бан от API за спам
// Лаги при вводе
```

**Решение:** Debouncer / Throttling

```swift
// ✅ Хорошо: запрос через 500ms после последнего ввода
private let searchExecutor = CancellableExecutor()

func textFieldDidChange(_ text: String) {
    searchExecutor.execute(delay: .milliseconds(500)) { [weak self] isCancelled in
        guard !isCancelled.isCancelled else { return }
        self?.presenter.searchButtonTapped(query: text)
    }
}
```

**Где встречается:** 90% учебных проектов

**Production fix:**
```swift
// Использовать Combine или RxSwift
searchTextField.publisher
    .debounce(for: .milliseconds(500), scheduler: RunLoop.main)
    .sink { [weak self] query in
        self?.search(query)
    }
    .store(in: &cancellables)
```

### 2. Нет Пагинации

**Проблема:** Загружаем все результаты сразу.

```swift
// ❌ Плохо: загружаем всё
iconService.searchIcons(query: "home") { icons in
    self.icons = icons  // 1000+ иконок
    tableView.reloadData()  // Лаги!
}

// Результат:
// - Долгая загрузка
// - Много трафика
// - Memory warning
```

**Решение:** Постраничная загрузка

```swift
// ✅ Хорошо: по 20 штук
iconService.searchIcons(query: "home", page: 1, limit: 20) { icons in
    self.icons.append(contentsOf: icons)
    tableView.reloadData()
}

// Бесконечная лента
func scrollViewDidScroll(_ scrollView: UIScrollView) {
    let offsetY = scrollView.contentOffset.y
    let contentHeight = scrollView.contentSize.height
    let height = scrollView.frame.size.height
    
    if offsetY > contentHeight - height - 100 {
        loadMorePage()
    }
}
```

**Где встречается:** 80% проектов без ментора

### 3. UI Не в Main Thread

**Проблема:** Обновление UI из background потока.

```swift
// ❌ Плохо
networkManager.request { result in
    // Этот код выполняется в background потоке!
    self.tableView.reloadData()  // Crash или глюки
    self.loadingView.isHidden = true  // Не обновится
}

// Результат:
// - Случайные краши
// - UI не обновляется
// - "UI API called from background thread" warning
```

**Решение:** Всегда main thread для UI

```swift
// ✅ Хорошо
networkManager.request { result in
    DispatchQueue.main.async {
        self.tableView.reloadData()
        self.loadingView.isHidden = true
    }
}

// Или в NetworkManager
func request<T: Decodable>(url: URL, completion: @escaping (Result<T, Error>) -> Void) {
    let task = session.dataTask(with: url) { data, response, error in
        DispatchQueue.main.async {  // ← Всегда в main
            // ... обработка
        }
    }
    task.resume()
}
```

**Где встречается:** 70% проектов

## 🏛️ Архитектурные Ошибки

### 4. Massive ViewController

**Проблема:** Вся логика в ViewController.

```swift
// ❌ Плохо: ViewController делает всё
class IconSearchViewController: UIViewController {
    
    // Сеть
    func loadData() { ... }
    
    // Кэш
    func cacheImage(_ image: UIImage) { ... }
    
    // Бизнес-логика
    func filterIcons(query: String) { ... }
    
    // UI
    func setupTableView() { ... }
    
    // 800+ строк кода
}
```

**Решение:** MVP

```swift
// ✅ Хорошо: логика в Presenter
class IconSearchViewController: UIViewController {
    private let presenter: IconSearchPresenterProtocol
    
    // Только UI
    func showIcons(_ viewModels: [IconViewModel]) { ... }
    func showLoading() { ... }
}

class IconSearchPresenter {
    // Бизнес-логика здесь
    func searchButtonTapped(query: String) { ... }
    func filterIcons() { ... }
}
```

**Где встречается:** 95% проектов без архитектуры

### 5. Domain Модели в View

**Проблема:** View знает про модели из Domain layer.

```swift
// ❌ Плохо
protocol IconSearchViewProtocol {
    func showIcons(_ icons: [Icon])  // ← Domain модель!
}

// View должен знать про Icon, IconDetail, и т.д.
// Нарушение Separation of Concerns
```

**Решение:** ViewModel для View

```swift
// ✅ Хорошо
protocol IconSearchViewProtocol {
    func showIcons(_ viewModels: [IconViewModel])  // ← ViewModel
}

struct IconViewModel {
    // Только то, что нужно View
    let iconImageURL: URL?
    let sizeText: String
    let tagsText: String
}
```

### 6. Нет Протоколов для Сервисов

**Проблема:** Concrete implementation вместо protocol.

```swift
// ❌ Плохо
class IconSearchPresenter {
    private let service: IconifyService  // ← Конкретный класс
    
    init() {
        self.service = IconifyService()  // ← Создание внутри
    }
}

// Проблемы:
// - Нельзя замокать для тестов
// - Нельзя заменить на другой сервис
// - Сильная связанность
```

**Решение:** Protocol + Dependency Injection

```swift
// ✅ Хорошо
class IconSearchPresenter {
    private let service: IconServiceProtocol  // ← Протокол
    
    init(service: IconServiceProtocol) {  // ← Внедрение
        self.service = service
    }
}

// В тестах:
let mockService = MockIconService()
let presenter = IconSearchPresenter(service: mockService)
```

## 💾 Ошибки Памяти

### 7. Retain Cycle в Closure

**Проблема:** Сильная ссылка на self в closure.

```swift
// ❌ Плохо
class IconSearchPresenter {
    var onComplete: () -> Void = {}
    
    init() {
        onComplete = {
            self.doSomething()  // ← Сильная ссылка!
        }
    }
}

// Результат:
// - Presenter не удаляется из памяти
// - Утечка памяти
// - Crash через 10-20 минут
```

**Решение:** Weak self

```swift
// ✅ Хорошо
class IconSearchPresenter {
    var onComplete: () -> Void = {}
    
    init() {
        onComplete = { [weak self] in
            self?.doSomething()  // ← Слабая ссылка
        }
    }
}
```

**Где встречается:** 60% проектов

---

### 8. Нет Ограничений у Кэша

**Проблема:** Кэш растёт бесконечно.

```swift
// ❌ Плохо
let cache = NSCache<NSString, UIImage>()
// Нет countLimit и totalCostLimit

// Результат:
// - Кэш занимает всю память
// - Memory warning
// - Crash приложения
```

**Решение:** Установить лимиты

```swift
// ✅ Хорошо
let cache = NSCache<NSString, UIImage>()
cache.countLimit = 100
cache.totalCostLimit = 50 * 1024 * 1024  // 50 MB
```

## 🎨 UI Ошибки

### 9. prepareForReuse Не Используется

**Проблема:** Картинка не сбрасывается при reuse ячейки.

```swift
// ❌ Плохо
class IconTableViewCell: UITableViewCell {
    
    func configure(with viewModel: IconViewModel) {
        // Загружаем картинку
        imageLoader.load(url: viewModel.iconURL) { image in
            self.imageView.image = image
        }
    }
    
    // Нет prepareForReuse!
}

// Результат:
// - При скролле появляются чужие картинки
// - "Мигание" ячеек
// - Неправильные данные
```

**Решение:** Сброс в prepareForReuse

```swift
// ✅ Хорошо
class IconTableViewCell: UITableViewCell {
    
    override func prepareForReuse() {
        super.prepareForReuse()
        imageView.image = nil  // Сброс картинки
        cellPresenter.cancelLoad()  // Отмена загрузки
    }
    
    func configure(with viewModel: IconViewModel) {
        cellPresenter.loadImage(from: viewModel.iconURL)
    }
}
```

**Где встречается:** 85% проектов с таблицами

### 10. Нет Состояний UI

**Проблема:** Только 2 состояния: загрузка и результаты.

```swift
// ❌ Плохо
enum UIState {
    case loading
    case results([Icon])
}

// Нет состояния для:
// - Пустой результат (ничего не найдено)
// - Ошибка сервера
// - Начальное состояние (ещё не искали)
```

**Решение:** Полная модель состояний

```swift
// ✅ Хорошо
enum UIState {
    case initial  // Ещё не искали
    case loading  // Загрузка
    case results([Icon])  // Есть результаты
    case empty  // Ничего не найдено
    case error(Error)  // Ошибка
}

// В View:
func render(state: UIState) {
    switch state {
    case .initial:
        showSearchPrompt()
    case .loading:
        showLoadingSpinner()
    case .results(let icons):
        showTableView(icons)
    case .empty:
        showEmptyStateLabel()
    case .error(let error):
        showErrorAlert(error.localizedDescription)
    }
}
```

## 💾 Ошибки Кэширования

### 11. UserDefaults для Кэша

**Проблема:** Кэширование картинок в UserDefaults.

```swift
// ❌ КАТЕГОРИЧЕСКИ НЕЛЬЗЯ
UserDefaults.standard.set(imageData, forKey: "cached_image")

// Почему плохо:
// - Блокирует главный поток
// - Пишет на диск каждый раз
// - Нет лимитов
// - Может вырасти до гигабайт
```

**Решение:** NSCache для памяти, FileManager для диска

```swift
// ✅ Хорошо
let cache = Cache<URL, UIImage>()
cache.totalCostLimit = 50 * 1024 * 1024
cache.setValue(image, forKey: url)
```

---

### 12. Dictionary Вместо NSCache

**Проблема:** Самодельный кэш на Dictionary.

```swift
// ❌ Плохо
var cache: [URL: UIImage] = [:]

// Проблемы:
// - Не потокобезопасный
// - Нет автоочистки при memory warning
// - Копирует объекты (медленнее)
```

**Решение:** NSCache

```swift
// ✅ Хорошо
let cache = NSCache<NSURL, UIImage>()
// или универсальный
let cache = Cache<URL, UIImage>()
```

## ✅ Чеклист Перед Сдачей

### Код

- [ ] Нет warnings при сборке
- [ ] Следование Swift API Design Guidelines
- [ ] Комментарии только для сложной логики
- [ ] Нет `print()` в production коде
- [ ] Weak self в closure где нужно
- [ ] UI обновления в main thread

### Функциональность

- [ ] Поиск работает
- [ ] Debouncer для поиска (500ms)
- [ ] Пагинация (бесконечная лента)
- [ ] Кэш картинок работает
- [ ] Кэш API запросов работает
- [ ] Empty state отображается
- [ ] Error state отображается
- [ ] Loading state отображается
- [ ] Сохранение в галерею работает

### UI

- [ ] prepareForReuse в ячейках
- [ ] Картинки не "мигают" при скролле
- [ ] Нет лагов при скролле
- [ ] Все состояния UI реализованы

### Тесты

- [ ] 2+ Unit теста
- [ ] 2+ UI теста
- [ ] Тесты проходят

### Архитектура

- [ ] MVP соблюдается
- [ ] View не знает про Domain модели
- [ ] Сервисы через протоколы
- [ ] Dependency Injection используется
