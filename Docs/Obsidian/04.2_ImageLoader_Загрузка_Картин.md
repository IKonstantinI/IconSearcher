## 🎯 Назначение ImageLoader

### Какую Проблему Решает

```
Проблема: Загрузка картинок в таблице требует:
├── Кэширование (не грузить одно и то же дважды)
├── Отмена загрузки (ячейка ушла в пул)
├── Обработка ошибок (нет сети, 404)
├── Главный поток (UI обновления)
└── Дедупликация запросов (10 ячеек, 1 картинка)

Решение: ImageLoader — централизованный сервис
```

### Зона Ответственности

| Делает | Не Делает |
|--------|-----------|
| ✅ Загрузка из сети | ❌ Не хранит данные навсегда |
| ✅ Кэширование в памяти | ❌ Не знает о UI (UITableViewCell) |
| ✅ Отмена по запросу | ❌ Не форматирует изображения |
| ✅ Callback с результатом | ❌ Не решает, когда загружать |

## 📜 Протокол ImageLoaderProtocol

### Назначение

**Для чего:** Абстракция над загрузкой картинок для:
- Внедрения зависимостей (DI)
- Тестирования (mock для unit тестов)
- Замены реализации (например, на Nuke/Kingfisher)

### Интерфейс

```swift
import UIKit

/// Протокол сервиса загрузки изображений.
/// 
/// Абстракция для загрузки и кэширования картинок.
/// Используется в Presenter для загрузки изображений иконок.
protocol ImageLoaderProtocol {
    
    /// Загрузить изображение по URL.
    /// - Parameters:
    ///   - url: URL изображения
    ///   - completion: Closure с результатом (UIImage? или Error)
    /// - Returns: Task ID для отмены загрузки
    func loadImage(
        from url: URL,
        completion: @escaping (UIImage?) -> Void
    ) -> Int
    
    /// Отменить загрузку изображения.
    /// - Parameter taskId: Task ID, возвращённый из loadImage
    func cancelLoad(taskId: Int)
    
    /// Очистить кэш изображений.
    func clearCache()
}
```

## 💻 Реализация ImageLoader

### Production Версия

```swift
import UIKit

/// Сервис загрузки изображений с кэшированием.
/// 
/// Использует двухуровневое кэширование:
/// 1. NSCache для мгновенного доступа
/// 2. URLCache для HTTP-ответов (автоматически)
/// 
/// Особенности:
/// - Потокобезопасный (все операции в background)
/// - Отмена загрузки по запросу
/// - Дедупликация запросов (одинаковые URL)
final class ImageLoader: ImageLoaderProtocol {
    
    // MARK: - Types
    
    /// Task для отслеживания загрузки
    private struct LoadTask {
        let url: URL
        let completion: (UIImage?) -> Void
    }
    
    // MARK: - Properties
    
    /// Кэш для хранения загруженных изображений в памяти
    private let cache: Cache<URL, UIImage>
    
    /// URLSession для сетевых запросов
    private let session: URLSession
    
    /// Активные задачи загрузки: [taskId: LoadTask]
    private var activeTasks: [Int: LoadTask] = [:]
    
    /// Счётчик для генерации уникальных task ID
    private var taskCounter: Int = 0
    
    /// Lock для синхронизации доступа к activeTasks
    private let lock = NSLock()
    
    // MARK: - Initialization
    
    /// Инициализация с кастомными параметрами.
    /// - Parameters:
    ///   - memoryCapacity: Максимальный размер кэша в байтах (по умолчанию 50 MB)
    ///   - countLimit: Максимальное количество изображений в кэше (по умолчанию 100)
    init(
        memoryCapacity: Int = 50 * 1024 * 1024,
        countLimit: Int = 100
    ) {
        self.cache = Cache()
        self.cache.totalCostLimit = memoryCapacity
        self.cache.countLimit = countLimit
        
        // URLSession с URLCache для HTTP-кэширования
        let config = URLSessionConfiguration.default
        config.urlCache = URLCache(
            memoryCapacity: 50 * 1024 * 1024,
            diskCapacity: 100 * 1024 * 1024
        )
        self.session = URLSession(configuration: config)
    }
    
    // MARK: - Public Methods
    
    func loadImage(
        from url: URL,
        completion: @escaping (UIImage?) -> Void
    ) -> Int {
        // 1. Проверяем кэш
        if let cachedImage = cache.value(forKey: url) {
            // ✅ Кэш найден — возвращаем сразу
            DispatchQueue.main.async {
                completion(cachedImage)
            }
            return -1 // Нет задачи, кэш
        }
        
        // 2. Проверяем, нет ли уже активной загрузки этого URL
        if let existingTaskId = findActiveTask(for: url) {
            // ✅ Уже загружается — подписываемся на тот же результат
            return existingTaskId
        }
        
        // 3. Создаём новую задачу
        let taskId = nextTaskId()
        let task = LoadTask(url: url, completion: completion)
        
        lock.lock()
        activeTasks[taskId] = task
        lock.unlock()
        
        // 4. Запускаем загрузку
        performLoad(task: task, taskId: taskId)
        
        return taskId
    }
    
    func cancelLoad(taskId: Int) {
        // Игнорируем -1 (кэш)
        guard taskId != -1 else { return }
        
        lock.lock()
        activeTasks.removeValue(forKey: taskId)
        lock.unlock()
    }
    
    func clearCache() {
        cache.removeAll()
    }
    
    // MARK: - Private Methods
    
    /// Найти активную задачу для URL
    private func findActiveTask(for url: URL) -> Int? {
        lock.lock()
        defer { lock.unlock() }
        
        for (taskId, task) in activeTasks where task.url == url {
            return taskId
        }
        return nil
    }
    
    /// Следующий ID задачи
    private func nextTaskId() -> Int {
        lock.lock()
        defer { lock.unlock() }
        taskCounter += 1
        return taskCounter
    }
    
    /// Выполнить загрузку
    private func performLoad(task: LoadTask, taskId: Int) {
        let dataTask = session.dataTask(with: task.url) { [weak self] data, response, error in
            guard let self = self else { return }
            
            // Проверяем, не отменили ли задачу
            self.lock.lock()
            let isActive = self.activeTasks[taskId] != nil
            self.lock.unlock()
            
            guard isActive else { return }
            
            // Обрабатываем результат
            var image: UIImage?
            
            if let data = data, error == nil {
                image = UIImage(data: data)
                // Сохраняем в кэш
                self.cache.setValue(image, forKey: task.url)
            }
            
            // Удаляем задачу
            self.lock.lock()
            let storedTask = self.activeTasks.removeValue(forKey: taskId)
            self.lock.unlock()
            
            // Вызываем completion в main thread
            DispatchQueue.main.async {
                storedTask?.completion(image)
            }
        }
        
        dataTask.resume()
    }
}
```

## 🔗 Интеграция с Кэшем

### Двухуровневое Кэширование

```
┌─────────────────────────────────────────────────────────────┐
│                    ImageLoader.loadImage()                  │
└───────────────────────────┬─────────────────────────────────┘
                            │
                            ▼
                 ┌─────────────────────┐
                 │  1. Cache<URL,UIImage> │ ← Проверяем сначала
                 │     (NSCache)          │
                 └───────────┬───────────┘
                             │
                    ┌────────┴────────┐
                    │                 │
               ✅ Есть          ❌ Нет
                    │                 │
                    ▼                 ▼
            Возвращаем       ┌─────────────────┐
            из кэша          │  2. URLSession   │ ← Загружаем
            (0.1ms)          │    (сеть)        │   из сети
                             └────────┬────────┘
                                      │
                                      ▼
                             ┌─────────────────┐
                             │  Сохраняем в    │
                             │  Cache          │
                             └─────────────────┘
```

### Преимущества

| Уровень | Скорость | Источник | Когда |
|---------|----------|----------|-------|
| **NSCache** | ~0.1ms | Память | Повторный доступ |
| **URLCache** | ~10ms | Диск | Повторный запрос после рестарта |
| **Сеть** | ~500ms | API | Первый запрос |

## 🎯 Использование в CellPresenter

### Схема

```
IconTableViewCell
       │
       │ configure(with viewModel:)
       ▼
IconCellPresenter
       │
       │ loadImage(from: URL)
       ▼
ImageLoaderProtocol
       │
       │ Загрузка + кэш
       ▼
Delegate: imageDidLoad(image)
       │
       ▼
IconTableViewCell: imageView.image = image
```

### Пример Кода

```swift
// В IconCellPresenter
final class IconCellPresenter {
    
    private weak var delegate: IconCellPresenterDelegate?
    private let imageLoader: ImageLoaderProtocol
    private var currentTaskId: Int = -1
    
    init(delegate: IconCellPresenterDelegate, imageLoader: ImageLoaderProtocol) {
        self.delegate = delegate
        self.imageLoader = imageLoader
    }
    
    func loadImage(from url: URL?) {
        guard let url = url else {
            delegate?.imageLoader(self, didLoad: nil)
            return
        }
        
        // Отменяем предыдущую загрузку
        cancelLoad()
        
        // Запускаем новую
        currentTaskId = imageLoader.loadImage(from: url) { [weak self] image in
            self?.delegate?.imageLoader(self!, didLoad: image)
        }
    }
    
    func cancelLoad() {
        imageLoader.cancelLoad(taskId: currentTaskId)
        currentTaskId = -1
    }
}
```

## ✅ Best Practices

### 1. Отмена в prepareForReuse

```swift
// В UITableViewCell
override func prepareForReuse() {
    super.prepareForReuse()
    cellPresenter.cancelLoad()  // Отменяем загрузку
    imageView.image = nil       // Сбрасываем
}
```

### 2. Weak Self в Closure

```swift
// ✅ Хорошо
imageLoader.loadImage(from: url) { [weak self] image in
    self?.imageView.image = image
}

// ❌ Плохо
imageLoader.loadImage(from: url) { image in
    self.imageView.image = image  // Retain cycle!
}
```

### 3. Главный Поток для UI

```swift
// ✅ Хорошо (в ImageLoader)
DispatchQueue.main.async {
    completion(image)
}

// ❌ Плохо
completion(image)  // Может быть в background thread!
```

### 4. Проверка на Активность

```swift
// ✅ Хорошо: проверяем, не отменили ли
self.lock.lock()
let isActive = self.activeTasks[taskId] != nil
self.lock.unlock()

guard isActive else { return }

// ❌ Плохо: вызываем completion всегда
completion(image)  // Даже если ячейка ушла в пул
```

## ⚠️ Антипаттерны

### 1. Загрузка в Ячейке Напрямую

```swift
// ❌ Плохо: ячейка знает про URLSession
class IconTableViewCell: UITableViewCell {
    func configure(url: URL) {
        URLSession.shared.dataTask(with: url) { ... }  // No!
    }
}

// ✅ Хорошо: через Presenter
class IconTableViewCell: UITableViewCell {
    func configure(presenter: IconCellPresenter) {
        // Presenter решает, когда и как загружать
    }
}
```

### 2. Нет Отмены Загрузки

```swift
// ❌ Плохо
func loadImage(from url: URL) {
    // Запускаем и забываем
    URLSession.shared.dataTask(with: url) { ... }
}

// ✅ Хорошо
func loadImage(from url: URL) -> Int {
    let taskId = nextId()
    // Сохраняем taskId для отмены
    return taskId
}

func cancelLoad(taskId: Int) {
    // Отменяем
}
```

### 3. Кэш Без Лимитов

```swift
// ❌ Плохо
private let cache = NSCache<NSURL, UIImage>()  // Нет лимитов!

// ✅ Хорошо
private let cache = Cache<URL, UIImage>()
cache.totalCostLimit = 50 * 1024 * 1024  // 50 MB
cache.countLimit = 100
```

### 4. UI Не в Main Thread

```swift
// ❌ Плохо
session.dataTask(with: url) { data, _, _ in
    let image = UIImage(data: data!)
    completion(image)  // В background thread!
}

// ✅ Хорошо
session.dataTask(with: url) { data, _, _ in
    let image = UIImage(data: data!)
    DispatchQueue.main.async {
        completion(image)  // В main thread
    }
}
```

## 🧪 Тестирование

### Mock для Тестов

```swift
import UIKit

/// Mock ImageLoader для unit тестов.
final class MockImageLoader: ImageLoaderProtocol {
    
    var loadImageCalled = false
    var cancelLoadCalled = false
    var clearCacheCalled = false
    
    var mockImage: UIImage?
    
    func loadImage(
        from url: URL,
        completion: @escaping (UIImage?) -> Void
    ) -> Int {
        loadImageCalled = true
        completion(mockImage)
        return 1
    }
    
    func cancelLoad(taskId: Int) {
        cancelLoadCalled = true
    }
    
    func clearCache() {
        clearCacheCalled = true
    }
}
```

### Unit Тест Пример

```swift
import XCTest

final class ImageLoaderTests: XCTestCase {
    
    var imageLoader: ImageLoader!
    
    func testLoadImageFromCache() {
        // Given
        let testImage = UIImage()
        let testURL = URL(string: "https://example.com/icon.png")!
        imageLoader = ImageLoader()
        imageLoader.cache.setValue(testImage, forKey: testURL)
        
        // When
        let expectation = XCTestExpectation(description: "Image loaded")
        imageLoader.loadImage(from: testURL) { image in
            // Then
            XCTAssertNotNil(image)
            XCTAssertEqual(image, testImage)
            expectation.fulfill()
        }
        
        wait(for: [expectation], timeout: 1.0)
    }
    
    func testCancelLoad() {
        // Given
        imageLoader = ImageLoader()
        let testURL = URL(string: "https://example.com/icon.png")!
        
        // When
        let taskId = imageLoader.loadImage(from: testURL) { _ in }
        imageLoader.cancelLoad(taskId: taskId)
        
        // Then
        // Задача должна быть отменена
    }
}
```
