## 🎯 Назначение TableViewCell

### Что Решает

```
Проблема: Отображение данных в таблице
├── Как показать иконку, размер, теги?
├── Как обработать reuse ячеек?
├── Как отменить загрузку при reuse?
└── Как обновить UI при загрузке картинки?

Решение: IconTableViewCell
- UI элементы (imageView, labels)
- prepareForReuse для сброса
- Delegate для получения картинки
```

### Зона Ответственности

| Делает | Не Делает |
|--------|-----------|
| ✅ Отображение UI элементов | ❌ Не содержит бизнес-логики |
| ✅ Вызов CellPresenter | ❌ Не загружает данные |
| ✅ prepareForReuse | ❌ Не знает про API |
| ✅ Delegate callback | ❌ Не маппит модели |

## 💻 Реализация IconTableViewCell

### Полная Версия (Programmatically)

```swift
import UIKit

/// Ячейка таблицы для отображения иконки.
/// 
/// Содержит:
/// - UIImageView для иконки
/// - UILabel для размера
/// - UILabel для тегов
/// 
/// Архитектура: MVP (View layer для ячейки)
final class IconTableViewCell: UITableViewCell {
    
    // MARK: - UI Elements
    
    private let iconImageView: UIImageView = {
        let imageView = UIImageView()
        imageView.contentMode = .scaleAspectFit
        imageView.clipsToBounds = true
        imageView.translatesAutoresizingMaskIntoConstraints = false
        return imageView
    }()
    
    private let sizeLabel: UILabel = {
        let label = UILabel()
        label.font = UIFont.systemFont(ofSize: 14, weight: .regular)
        label.textColor = .secondaryLabel
        label.translatesAutoresizingMaskIntoConstraints = false
        return label
    }()
    
    private let tagsLabel: UILabel = {
        let label = UILabel()
        label.font = UIFont.systemFont(ofSize: 12, weight: .regular)
        label.textColor = .tertiaryLabel
        label.numberOfLines = 2
        label.translatesAutoresizingMaskIntoConstraints = false
        return label
    }()
    
    private let loadingIndicator: UIActivityIndicatorView = {
        let indicator = UIActivityIndicatorView(style: .medium)
        indicator.hidesWhenStopped = true
        indicator.translatesAutoresizingMaskIntoConstraints = false
        return indicator
    }()
    
    // MARK: - Properties
    
    /// Presenter для этой ячейки
    private var cellPresenter: IconCellPresenter?
    
    /// Identifier для reuseIdentifier
    static let reuseIdentifier = String(describing: IconTableViewCell.self)
    
    // MARK: - Initialization
    
    override init(style: UITableViewCell.CellStyle, reuseIdentifier: String?) {
        super.init(style: style, reuseIdentifier: reuseIdentifier)
        setupUI()
        setupConstraints()
    }
    
    required init?(coder: NSCoder) {
        fatalError("init(coder:) has not been implemented")
    }
    
    // MARK: - Lifecycle
    
    override func prepareForReuse() {
        super.prepareForReuse()
        
        // 1. Отменяем загрузку изображения
        cellPresenter?.cancelLoad()
        
        // 2. Сбрасываем изображение
        iconImageView.image = nil
        
        // 3. Сбрасываем текстовые данные
        sizeLabel.text = nil
        tagsLabel.text = nil
        
        // 4. Показываем индикатор загрузки
        loadingIndicator.startAnimating()
    }
    
    // MARK: - UI Setup
    
    private func setupUI() {
        contentView.addSubview(iconImageView)
        contentView.addSubview(sizeLabel)
        contentView.addSubview(tagsLabel)
        contentView.addSubview(loadingIndicator)
        
        // Настройка imageView
        iconImageView.backgroundColor = .systemGray6
    }
    
    private func setupConstraints() {
        NSLayoutConstraint.activate([
            // Icon Image View
            iconImageView.leadingAnchor.constraint(
                equalTo: contentView.leadingAnchor, constant: 16),
            iconImageView.centerYAnchor.constraint(
                equalTo: contentView.centerYAnchor),
            iconImageView.widthAnchor.constraint(equalToConstant: 48),
            iconImageView.heightAnchor.constraint(equalToConstant: 48),
            
            // Loading Indicator (centered on imageView)
            loadingIndicator.centerXAnchor.constraint(
                equalTo: iconImageView.centerXAnchor),
            loadingIndicator.centerYAnchor.constraint(
                equalTo: iconImageView.centerYAnchor),
            
            // Size Label
            sizeLabel.topAnchor.constraint(
                equalTo: contentView.topAnchor, constant: 16),
            sizeLabel.leadingAnchor.constraint(
                equalTo: iconImageView.trailingAnchor, constant: 16),
            sizeLabel.trailingAnchor.constraint(
                equalTo: contentView.trailingAnchor, constant: -16),
            
            // Tags Label
            tagsLabel.topAnchor.constraint(
                equalTo: sizeLabel.bottomAnchor, constant: 4),
            tagsLabel.leadingAnchor.constraint(
                equalTo: iconImageView.trailingAnchor, constant: 16),
            tagsLabel.trailingAnchor.constraint(
                equalTo: contentView.trailingAnchor, constant: -16),
            tagsLabel.bottomAnchor.constraint(
                equalTo: contentView.bottomAnchor, constant: -16)
        ])
    }
    
    // MARK: - Public Methods
    
    /// Настроить ячейку с данными.
    /// - Parameters:
    ///   - viewModel: ViewModel с данными для отображения
    ///   - presenter: Presenter для загрузки изображения
    func configure(
        with viewModel: IconViewModel,
        presenter: IconCellPresenter
    ) {
        // 1. Сохраняем presenter
        self.cellPresenter = presenter
        
        // 2. Настраиваем текстовые данные
        sizeLabel.text = viewModel.sizeText
        tagsLabel.text = viewModel.tagsText
        
        // 3. Скрываем индикатор загрузки
        loadingIndicator.stopAnimating()
        
        // 4. Загружаем изображение через presenter
        presenter.loadImage(from: viewModel.iconImageURL)
    }
}
```

## 🔄 prepareForReuse

### Зачем Нужно

```
┌─────────────────────────────────────────────────────────────┐
│  Проблема без prepareForReuse                              │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│  1. Ячейка 1 запрашивает картинку /mdi/home.svg            │
│     Загрузка началась (500ms)                               │
│                                                             │
│  2. Пользователь скроллит                                   │
│     Ячейка 1 уходит в пул                                  │
│                                                             │
│  3. Пользователь скроллит обратно                           │
│     Ячейка 1 появляется для индекса 5                      │
│     Данные: /fa/user.svg                                   │
│                                                             │
│  4. Загрузка из шага 1 завершается                          │
│     /mdi/home.svg устанавливается в imageView              │
│                                                             │
│  Результат: Неправильная картинка!                          │
│                                                             │
└─────────────────────────────────────────────────────────────┘

┌─────────────────────────────────────────────────────────────┐
│  Решение с prepareForReuse                                 │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│  override func prepareForReuse() {                          │
│      super.prepareForReuse()                                │
│      cellPresenter?.cancelLoad()  // ← Отмена загрузки!     │
│      iconImageView.image = nil    // ← Сброс image!         │
│  }                                                          │
│                                                             │
│  Результат:                                                 │
│  - Старая загрузка отменена                                 │
│  - Ячейка чистая перед настройкой                           │
│  - Правильная картинка загружается                          │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

### Реализация

```swift
override func prepareForReuse() {
    super.prepareForReuse()
    
    // 1. Отменяем загрузку изображения
    // Это критично! Иначе старая загрузка завершится
    // и перезапишет правильную картинку
    cellPresenter?.cancelLoad()
    
    // 2. Сбрасываем изображение
    // Ячейка должна быть "чистой" перед повторным использованием
    iconImageView.image = nil
    
    // 3. Сбрасываем текстовые данные
    // Иначе старые данные будут видны до настройки новых
    sizeLabel.text = nil
    tagsLabel.text = nil
    
    // 4. Показываем индикатор загрузки
    // Пользователь видит, что данные загружаются
    loadingIndicator.startAnimating()
}
```

## 🔗 Интеграция с CellPresenter

### Схема Взаимодействия

```
┌─────────────────────────────────────────────────────────────┐
│  1. ViewController создаёт ячейку                          │
│                                                             │
│  func tableView(_ tableView: UITableView,                   │
│                 cellForRowAt indexPath: IndexPath)          │
│         -> UITableViewCell {                                │
│                                                             │
│      let cell = tableView.dequeueReusableCell(...)          │
│                                                             │
└─────────────────────────────────────────────────────────────┘
                            │
                            ▼
┌─────────────────────────────────────────────────────────────┐
│  2. ViewController настраивает ячейку                      │
│                                                             │
│      let viewModel = viewModels[indexPath.row]              │
│      let presenter = cellPresenters[indexPath.row]          │
│                                                             │
│      cell.configure(with: viewModel, presenter: presenter)  │
│                                                             │
└─────────────────────────────────────────────────────────────┘
                            │
                            ▼
┌─────────────────────────────────────────────────────────────┐
│  3. CellPresenter загружает изображение                    │
│                                                             │
│  func configure(with viewModel, presenter) {                │
│      self.cellPresenter = presenter                         │
│      presenter.loadImage(from: viewModel.iconImageURL)      │
│  }                                                          │
│                                                             │
└─────────────────────────────────────────────────────────────┘
                            │
                            ▼
┌─────────────────────────────────────────────────────────────┐
│  4. ImageLoader загружает из кэша или сети                 │
│                                                             │
│  imageLoader.loadImage(from: url) { image in               │
│      // Кэш или сеть                                        │
│  }                                                          │
│                                                             │
└─────────────────────────────────────────────────────────────┘
                            │
                            ▼
┌─────────────────────────────────────────────────────────────┐
│  5. CellPresenter уведомляет делегата                      │
│                                                             │
│  delegate?.imageLoader(self, didLoad: image)                │
│                                                             │
└─────────────────────────────────────────────────────────────┘
                            │
                            ▼
┌─────────────────────────────────────────────────────────────┐
│  6. Ячейка обновляет UI                                    │
│                                                             │
│  func imageLoader(_ presenter, didLoad image: UIImage?) {   │
│      iconImageView.image = image                            │
│  }                                                          │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

### Реализация Delegate

```swift
// MARK: - IconCellPresenterDelegate

extension IconTableViewCell: IconCellPresenterDelegate {
    
    func imageLoader(
        _ presenter: IconCellPresenter,
        didLoad image: UIImage?
    ) {
        // Проверяем, что ячейка всё ещё актуальна
        // (не была переиспользована для других данных)
        
        // Устанавливаем изображение
        iconImageView.image = image
        
        // Скрываем индикатор загрузки
        loadingIndicator.stopAnimating()
        
        // Анимация появления (опционально)
        if image != nil {
            UIView.transition(
                with: iconImageView,
                duration: 0.2,
                options: .transitionCrossDissolve,
                animations: {}
            )
        }
    }
    
    func imageLoader(
        _ presenter: IconCellPresenter,
        didFailWith error: Error
    ) {
        // Показываем placeholder при ошибке
        iconImageView.image = UIImage(named: "icon_placeholder")
        
        // Скрываем индикатор
        loadingIndicator.stopAnimating()
        
        // Логирование (опционально)
        print("IconTableViewCell: Failed to load image: \(error.localizedDescription)")
    }
}
```

## ✅ Best Practices

### 1. Programmatically UI

```swift
// ✅ Хорошо: программно
private let iconImageView: UIImageView = {
    let imageView = UIImageView()
    imageView.contentMode = .scaleAspectFit
    return imageView
}()

// ❌ Плохо: Storyboard/XIB
// - Сложнее version control
// - Merge conflicts
// - Меньше контроля
```

### 2. Static Reuse Identifier

```swift
// ✅ Хорошо
static let reuseIdentifier = String(describing: IconTableViewCell.self)

// Использование:
tableView.register(
    IconTableViewCell.self,
    forCellReuseIdentifier: IconTableViewCell.reuseIdentifier
)

// ❌ Плохо
let reuseIdentifier = "IconCell"  // Magic string
```

### 3. Сброс Данных в prepareForReuse

```swift
// ✅ Хорошо
override func prepareForReuse() {
    super.prepareForReuse()
    cellPresenter?.cancelLoad()
    iconImageView.image = nil
    sizeLabel.text = nil
    tagsLabel.text = nil
}

// ❌ Плохо
override func prepareForReuse() {
    super.prepareForReuse()
    // Только image сбрасывается!
    // Текст остаётся от старой ячейки
}
```

### 4. Weak Delegate в Presenter

```swift
// В CellPresenter (не в ячейке!)
final class IconCellPresenter {
    private weak var delegate: IconCellPresenterDelegate?
}
```

## ⚠️ Антипаттерны

### 1. Загрузка Картинки в Ячейке

```swift
// ❌ Плохо: ячейка загружает сама
final class IconTableViewCell: UITableViewCell {
    
    func configure(url: URL) {
        URLSession.shared.dataTask(with: url) { data, _, _ in
            let image = UIImage(data: data!)
            DispatchQueue.main.async {
                self.iconImageView.image = image
            }
        }.resume()
    }
}

// ✅ Хорошо: через Presenter
final class IconTableViewCell: UITableViewCell {
    
    func configure(viewModel: IconViewModel, presenter: IconCellPresenter) {
        presenter.loadImage(from: viewModel.iconImageURL)
    }
}
```

---

### 2. Нет prepareForReuse

```swift
// ❌ Плохо
final class IconTableViewCell: UITableViewCell {
    // Нет prepareForReuse!
    
    func configure(viewModel: IconViewModel, presenter: IconCellPresenter) {
        // ...
    }
}

// Результат:
// - Старые данные видны до загрузки новых
// - Картинка из старой ячейки появляется
// - "Мигание" при скролле

// ✅ Хорошо
final class IconTableViewCell: UITableViewCell {
    
    override func prepareForReuse() {
        super.prepareForReuse()
        cellPresenter?.cancelLoad()
        iconImageView.image = nil
        sizeLabel.text = nil
        tagsLabel.text = nil
    }
}
```

---

### 3. Сильная Ссылка в Closure

```swift
// ❌ Плохо
func imageLoader(_ presenter: IconCellPresenter, didLoad image: UIImage?) {
    // Если бы был closure:
    someAsyncMethod {
        self.iconImageView.image = image  // Retain cycle!
    }
}

// ✅ Хорошо
func imageLoader(_ presenter: IconCellPresenter, didLoad image: UIImage?) {
    someAsyncMethod { [weak self] in
        self?.iconImageView.image = image
    }
}
```

---

### 4. UI Обновление Не в Main Thread

```swift
// ❌ Плохо
func imageLoader(_ presenter: IconCellPresenter, didLoad image: UIImage?) {
    // Если вызван из background thread:
    iconImageView.image = image  // Crash!
}

// ✅ Хорошо
func imageLoader(_ presenter: IconCellPresenter, didLoad image: UIImage?) {
    // ImageLoader уже возвращает в main thread
    iconImageView.image = image
}
```

## 🧪 Тестирование

### UI Тест Пример

```swift
import XCTest

final class IconTableViewCellTests: XCTestCase {
    
    var cell: IconTableViewCell!
    var mockPresenter: MockCellPresenter!
    
    override func setUp() {
        super.setUp()
        cell = IconTableViewCell()
        mockPresenter = MockCellPresenter()
    }
    
    func testConfigure_SetsTextData() {
        // Given
        let viewModel = IconViewModel(
            iconImageURL: URL(string: "https://example.com/icon.png"),
            sizeText: "24x24",
            tagsText: "Tags: home, house"
        )
        
        // When
        cell.configure(with: viewModel, presenter: mockPresenter)
        
        // Then
        XCTAssertEqual(cell.sizeLabel.text, "24x24")
        XCTAssertEqual(cell.tagsLabel.text, "Tags: home, house")
    }
    
    func testPrepareForReuse_ResetsData() {
        // Given
        let viewModel = IconViewModel(
            iconImageURL: URL(string: "https://example.com/icon.png"),
            sizeText: "24x24",
            tagsText: "Tags: home"
        )
        cell.configure(with: viewModel, presenter: mockPresenter)
        
        // When
        cell.prepareForReuse()
        
        // Then
        XCTAssertNil(cell.iconImageView.image)
        XCTAssertNil(cell.sizeLabel.text)
        XCTAssertNil(cell.tagsLabel.text)
        XCTAssertTrue(cell.loadingIndicator.isAnimating)
    }
}
```
